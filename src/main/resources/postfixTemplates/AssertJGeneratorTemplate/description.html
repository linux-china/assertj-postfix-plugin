Add following code in pom.xml: 

<plugin>
    <groupId>org.assertj</groupId>
    <artifactId>assertj-assertions-generator-maven-plugin</artifactId>
    <version>2.0.0</version>
    <configuration>
        <entryPointClassPackage>org.assertj</entryPointClassPackage>
        <classes>
            <param>
            org.mvnsearch.domain.model.account.Account</param>
            <param>
            org.mvnsearch.domain.model.parcel.Parcel</param>
        </classes>
        <generateAssertions>true</generateAssertions>
        <generateBddAssertions>false</generateBddAssertions>
        <generateSoftAssertions>false</generateSoftAssertions>
        <generateJUnitSoftAssertions>false</generateJUnitSoftAssertions>
    </configuration>
    <executions>
        <execution>
            <phase>compile</phase>
            <goals>
                <goal>generate-assertions</goal>
            </goals>
        </execution>
    </executions>
</plugin>
